\section*{Part I: Multiple Choice Questions (20 marks)}
Choose the correct answer for each question. Each question carries 2 marks.

\begin{enumerate}
    \item Which expression is equivalent to $(3x^2 y^{-3})^2$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $6x^4 y^{-6}$
        \item $9x^4 y^{-6}$
        \item $9x^4 y^{-5}$
        \item $6x^4 y^{-5}$
    \end{enumerate}
    \end{multicols}

    \item The discriminant of $x^2 - 4x + k = 0$ is 0. What is the value of $k$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item 2
        \item 4
        \item 6
        \item 8
    \end{enumerate}
    \end{multicols}

    \item Which inequality represents the solution set for $x^2 - 5x - 14 > 0$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $x < -2$ or $x > 7$
        \item $x < -7$ or $x > 2$
        \item $-2 < x < 7$
        \item $-7 < x < 2$
    \end{enumerate}
    \end{multicols}

    \item The graph of $y = \frac{1}{x}$ is translated by the vector $\begin{pmatrix} 2 \\ -3 \end{pmatrix}$. What is the equation of the new graph?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $y = \frac{1}{x-2} - 3$
        \item $y = \frac{1}{x+2} + 3$
        \item $y = \frac{1}{x-2} + 3$
        \item $y = \frac{1}{x+2} - 3$
    \end{enumerate}
    \end{multicols}

    \item Which of the following is a factor of $x^3 - 4x$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $x - 2$
        \item $x + 1$
        \item $x^2 + 4$
        \item $x - 4$
    \end{enumerate}
    \end{multicols}

    \item Simplify $\sqrt{12} + \sqrt{27}$.
    \begin{multicols}{4}
    \begin{enumerate}
        \item $5\sqrt{3}$
        \item $6\sqrt{3}$
        \item $7\sqrt{3}$
        \item $8\sqrt{3}$
    \end{enumerate}
    \end{multicols}

    \item What is the value of $8^{-\frac{2}{3}}$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $\frac{1}{4}$
        \item $\frac{1}{2}$
        \item 2
        \item 4
    \end{enumerate}
    \end{multicols}

    \item The function $f(x) = x^2 + 2x + 3$ has:
    \begin{multicols}{4}
    \begin{enumerate}
        \item Two distinct real roots
        \item One repeated real root
        \item No real roots
        \item A minimum point at $(-1, 2)$
    \end{enumerate}
    \end{multicols}

    \item Which of the following is the completed square form of $x^2 + 8x + 1$?
    \begin{multicols}{4}
    \begin{enumerate}
        \item $(x+4)^2 - 15$
        \item $(x+4)^2 + 15$
        \item $(x+8)^2 - 63$
        \item $(x+8)^2 + 1$
    \end{enumerate}
    \end{multicols}

    \item The solutions to the equation $x^2 - 6x + 9 = 0$ are:
    \begin{multicols}{4}
    \begin{enumerate}
        \item $x = 3$ only
        \item $x = -3$ only
        \item $x = 3$ and $x = -3$
        \item No real solutions
    \end{enumerate}
    \end{multicols}
\end{enumerate}

\section*{Part II: True/False Questions (10 marks)}
State whether each statement is True or False. Each question carries 2 marks.

\begin{enumerate}
    \item \textbf{True/False:} The function $f(x) = x^2 + 2x + 3$ has no real roots.
    \item \textbf{True/False:} The graph of $y = (x-2)^2 + 3$ has a minimum point at $(2, 3)$.
    \item \textbf{True/False:} The expression $\sqrt{12} + \sqrt{27}$ simplifies to $5\sqrt{3}$.
    \item \textbf{True/False:} The inequality $\frac{1}{x} > 2$ holds for all $x > 0$.
    \item \textbf{True/False:} The simultaneous equations $y = x^2$ and $y = 2x - 1$ have exactly one solution.
\end{enumerate}

\section*{Section D: Long Questions (40 marks)}
Show all steps clearly. Each question carries 10 marks.

\begin{enumerate}
    \item 
    \begin{enumerate}
        \item Solve the equation $x^2 - 10x + 16 = 0$ by factorisation.
        \item Hence, solve the inequality $x^2 - 10x + 16 \leq 0$.
        \item Sketch the graph of $y = x^2 - 10x + 16$, indicating key points.
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item Simplify $\frac{2x^2 - 8}{x^2 - 4x + 4}$.
        \item Express $\frac{3}{\sqrt{5} - 1}$ in the form $a + b\sqrt{5}$, where $a$ and $b$ are rational.
        \item Factorise completely: $x^3 - 3x^2 - 4x + 12$.
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item Sketch the graph of $y = x^2 - 4x + 3$, labeling intercepts and the turning point.
        \item On the same axes, sketch the graph of $y = x(x-1)(x+1)$, labeling intercepts.
        \item What is the vector that translates $y = x^2$ to $y = (x-1)^2 + 2$?
    \end{enumerate}

    \item 
    \begin{enumerate}
        \item Sketch the graphs of the following functions, labeling X and Y intercepts:
        \[
        \begin{cases}
        f(x) = (2+x)^2(2-x) \\
        g(x) = \frac{-4}{x^2}
        \end{cases}
        \]
        \item State, with reason, the number of solutions to the equation $f(x) = g(x)$.
        \item Using the sketch in part (a), shade the region that satisfies the inequalities $y\geq f(x)$ and $y\leq g(x)$.
    \end{enumerate}
\end{enumerate}
