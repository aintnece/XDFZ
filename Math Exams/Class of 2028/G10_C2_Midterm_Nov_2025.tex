\section*{Multiple Choice Questions (10 questions, 2 marks each)}
Circle the correct answer for each question.

\begin{enumerate}
\item Simplify $\dfrac{x^3 \times x^4}{x^{-2}}$
\begin{enumerate}
\item $x^5$
\item $x^9$ 
\item $x^{-1}$
\item $x^{24}$
\end{enumerate}

\item Which expression is equivalent to $(2x-3)(x+4)$?
\begin{enumerate}
\item $2x^2+5x-12$
\item $2x^2+11x-12$
\item $2x^2-5x-12$
\item $2x^2-11x-12$
\end{enumerate}

\item Factorize completely: $x^3-4x$
\begin{enumerate}
\item $x(x-2)(x+2)$
\item $x(x-4)$
\item $(x^2-2)(x+2)$
\item $x(x-2)^2$
\end{enumerate}

\item Simplify $\sqrt{75} + 2\sqrt{12} - \sqrt{27}$
\begin{enumerate}
\item $4\sqrt{3}$
\item $6\sqrt{3}$
\item $8\sqrt{3}$
\item $10\sqrt{3}$
\end{enumerate}

\item Solve the equation: $x^2-5x-14=0$
\begin{enumerate}
\item $x=2,7$
\item $x=-2,7$
\item $x=2,-7$
\item $x=-2,-7$
\end{enumerate}

\item The quadratic function $f(x)=2x^2-8x+5$ can be written in completed square form as:
\begin{enumerate}
\item $2(x-2)^2-3$
\item $2(x-4)^2-11$
\item $(x-4)^2-11$
\item $2(x-2)^2+13$
\end{enumerate}

\item Solve the simultaneous equations: 
$$y=2x-1$$
$$x^2+y^2=10$$
\begin{enumerate}
\item $(1,1)$ and $(-3,-7)$
\item $(1,1)$ and $(3,5)$
\item $(-1,-3)$ and $(3,5)$
\item $(2,3)$ and $(-2,-5)$
\end{enumerate}

\item Find the solution set for the inequality: $x^2-3x-10<0$
\begin{enumerate}
\item $\{x:-2<x<5\}$
\item $\{x:x<-2\}\cup\{x:x>5\}$
\item $\{x:-5<x<2\}$
\item $\{x:x<-5\}\cup\{x:x>2\}$
\end{enumerate}

\item The graph of $y=x^2$ is translated by the vector $\begin{pmatrix} 2 \\ -3 \end{pmatrix}$. What is the equation of the transformed graph?
\begin{enumerate}
\item $y=(x+2)^2-3$
\item $y=(x-2)^2-3$
\item $y=(x+3)^2+2$
\item $y=(x-3)^2+2$
\end{enumerate}

\item A straight line passes through points $(1,4)$ and $(3,10)$. What is its gradient?
\begin{enumerate}
\item 2
\item 3
\item 4
\item 6
\end{enumerate}
\end{enumerate}

\item $180^\circ$ is equal to $\pi$ radians.\hfill True \quad False
\end{enumerate}
